    **1. Create `components/animation/useFadeUp.ts`:**
    ```typescript
    'use client'

    import { gsap, ScrollTrigger, useGSAP } from '@/lib/gsap'

    export function useFadeUp(containerRef: React.RefObject<HTMLElement | null>) {
      useGSAP(
        () => {
          if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return

          const targets = containerRef.current?.querySelectorAll('[data-fade-up]') ?? []

          targets.forEach((el) => {
            gsap.set(el, { opacity: 0, y: 40 })

            ScrollTrigger.create({
              trigger: el,
              start: 'top 90%',
              end: 'bottom top',
              onEnter: () =>
                gsap.to(el, { opacity: 1, y: 0, duration: 0.7, ease: 'power2.out' }),
              onLeave: () =>
                gsap.to(el, { opacity: 0, y: -20, duration: 0.4, ease: 'power2.in' }),
              onEnterBack: () =>
                gsap.to(el, { opacity: 1, y: 0, duration: 0.7, ease: 'power2.out' }),
              onLeaveBack: () =>
                gsap.to(el, { opacity: 0, y: 40, duration: 0.4, ease: 'power2.in' }),
            })
          })
        },
        { scope: containerRef }
      )
    }
    ```

    - `start: 'top 90%'` = early trigger (D-04)
    - `end: 'bottom top'` = exit when element leaves viewport top
    - All four callbacks: re-animate each time (D-03)
    - Enter: 0.7s power2.out (UI-SPEC), Exit: 0.4s (faster exit feels snappy)
    - `gsap.set()` inside `useGSAP` to ensure CSS initial state and GSAP initial state are in sync
    - `{ scope: containerRef }` — useGSAP auto-kills all ScrollTriggers on unmount

    **3. Create `components/animation/useStagger.ts`:**
    ```typescript
    'use client'

    import { gsap, ScrollTrigger, useGSAP } from '@/lib/gsap'

    export function useStagger(containerRef: React.RefObject<HTMLElement | null>) {
      useGSAP(
        () => {
          if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return

          const containers = containerRef.current?.querySelectorAll('[data-stagger]') ?? []

          containers.forEach((container) => {
            const children = container.querySelectorAll('[data-stagger-child]')
            if (children.length === 0) return

            gsap.set(children, { opacity: 0, y: 40 })

            ScrollTrigger.create({
              trigger: container,
              start: 'top 90%',
              end: 'bottom top',
              onEnter: () =>
                gsap.to(children, {
                  opacity: 1,
                  y: 0,
                  duration: 0.7,
                  ease: 'power2.out',
                  stagger: {
                    each: 0.1,
                    ease: 'power1.out',
                    from: 'start',
                  },
                }),
              onLeave: () =>
                gsap.to(children, { opacity: 0, y: -20, duration: 0.3 }),
              onEnterBack: () =>
                gsap.to(children, {
                  opacity: 1,
                  y: 0,
                  duration: 0.7,
                  ease: 'power2.out',
                  stagger: {
                    each: 0.1,
                    ease: 'power1.out',
                    from: 'start',
                  },
                }),
              onLeaveBack: () =>
                gsap.to(children, { opacity: 0, y: 40, duration: 0.3 }),
            })
          })
        },
        { scope: containerRef }
      )
    }
    ```

    - Trigger is the `[data-stagger]` parent container, children are `[data-stagger-child]` elements
    - `stagger.each: 0.1` = 100ms between items (midpoint of D-02's 80-120ms range)
    - `stagger.ease: 'power1.out'` = eased cascade (discretion)
    - Exit: all children fade out simultaneously (0.3s) — fast exit to clear the stage

    ```typescript
    'use client'

    import { gsap, ScrollTrigger, useGSAP } from '@/lib/gsap'

    export function useHeroParallax(
      heroRef: React.RefObject<HTMLElement | null>,
      bgRef: React.RefObject<HTMLElement | null>,
      textRef: React.RefObject<HTMLElement | null>,
    ) {
      useGSAP(
        () => {
          if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return

          const hero = heroRef.current
          if (!hero) return

          // D-15: Parallax completely disabled on mobile
          if (window.innerWidth < 1024) {
            // D-14: Mobile fallback — simple fade-in, 500ms, no scroll effects
            gsap.fromTo(
              hero,
              { opacity: 0 },
              { opacity: 1, duration: 0.5, ease: 'power2.out' },
            )
            return
          }

          const heroHeight = hero.offsetHeight

          // D-12: Background parallax — 15% speed differential (midpoint of 10-20%)
          if (bgRef.current) {
            gsap.to(bgRef.current, {
              y: heroHeight * 0.15,
              ease: 'none',
              scrollTrigger: {
                trigger: hero,
                start: 'top top',
                end: 'bottom top',
                scrub: true,
              },
            })
          }

          // D-13: Hero text exit — fade out + drift up as user scrolls past hero
          if (textRef.current) {
            gsap.to(textRef.current, {
              opacity: 0,
              y: -30,
              ease: 'none',
              scrollTrigger: {
                trigger: hero,
                start: 'bottom 80%',
                end: 'bottom 20%',
                scrub: true,
              },
            })
          }

          // Handle resize — recalculate parallax travel distance
          const onResize = () => ScrollTrigger.refresh()
          window.addEventListener('resize', onResize)

          return () => {
            window.removeEventListener('resize', onResize)
          }
        },
        { scope: heroRef }
      )
    }
    ```

    Key decisions:
    - Accepts three refs: heroRef (container/trigger), bgRef (background image element), textRef (text content + CTA element)
    - `window.innerWidth < 1024` check — desktop-only guard (D-15). Uses 1024px which is the `lg` Tailwind breakpoint
    - Mobile fallback: simple `gsap.fromTo` opacity 0 to 1, 500ms (D-14). No scroll effects
    - Background `y` travels `heroHeight * 0.15` — 15% speed differential (D-12 midpoint)
    - Background `scrub: true` — scroll-linked, smooth, continuous (not time-based)
    - Text exit: opacity fades to 0, y drifts to -30px, scrub-linked to scroll position (D-13)
    - Text exit scroll range: hero bottom at 80% viewport (start) to hero bottom at 20% viewport (end) — discretion
    - `ScrollTrigger.refresh()` on window resize to recalculate cached trigger positions (Pitfall 5)
    - Resize listener cleaned up via returned cleanup function inside `useGSAP`
    - `{ scope: heroRef }` — all GSAP animations scoped and auto-cleaned on unmount
