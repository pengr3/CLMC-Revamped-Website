**Step 1: Scaffold the project**

Run the create-next-app command from the project root directory. The project files should be created directly in the current working directory (not in a subdirectory):

```bash
npx create-next-app@latest . --typescript --eslint --tailwind --app --no-src-dir --import-alias "@/*" --yes
```

If the directory is not empty (CLAUDE.md and .planning exist), the scaffold may refuse. In that case, scaffold into a temp directory and move files:
```bash
npx create-next-app@latest clmc-temp --typescript --eslint --tailwind --app --no-src-dir --import-alias "@/*" --yes
```
Then move all generated files (except .git if present) from clmc-temp/ to the project root, and delete clmc-temp/.

**Step 2: Install additional dependencies**

```bash
npm install gsap @gsap/react motion lenis clsx tailwind-merge lucide-react
```

**Step 3: Configure next.config.ts**

Replace the generated `next.config.ts` with:

```ts
import type { NextConfig } from 'next'

const nextConfig: NextConfig = {
  images: {
    formats: ['image/avif', 'image/webp'],
    deviceSizes: [375, 640, 768, 1024, 1280, 1920],
  },
}

export default nextConfig
```

Do NOT set `output: 'export'` — this breaks next/image optimization on Vercel.

/* ── Spacing scale ── */
@theme {
  --spacing-xs: 4px;
  --spacing-sm: 8px;
  --spacing-md: 16px;
  --spacing-lg: 24px;
  --spacing-xl: 32px;
  --spacing-2xl: 48px;
  --spacing-3xl: 64px;
  --spacing-4xl: 96px;
  --spacing-5xl: 128px;

/* ── Border radius tokens ── */

  --radius-sm: 4px;
  --radius-md: 8px;
  --radius-lg: 12px;
  --radius-full: 9999px;
}

/* ── Breakpoints (hardcoded rem — CSS vars NOT allowed in @media) ── */

  --breakpoint-sm: 23.4375rem;
  --breakpoint-md: 48rem;
  --breakpoint-lg: 64rem;
  --breakpoint-xl: 80rem;

    color: #F5F5F5;
  }

  }

Create `components/ui/Button.tsx` with the following exact implementation per the UI-SPEC Button contract:

```tsx
import { cn } from '@/lib/utils'
import { forwardRef } from 'react'

type ButtonVariant = 'primary' | 'ghost' | 'destructive'
type ButtonSize = 'sm' | 'md' | 'lg'

interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant
  size?: ButtonSize
}

const variantClasses: Record<ButtonVariant, string> = {
  primary: 'bg-text-primary text-surface-primary hover:bg-[rgba(245,245,245,0.85)]',
  ghost: 'bg-transparent text-text-primary border border-white/20 hover:border-white/60',
  destructive: 'bg-destructive text-white hover:bg-[#DC2626]',
}

const sizeClasses: Record<ButtonSize, string> = {
  sm: 'px-md py-sm text-[14px] rounded-sm',
  md: 'px-lg py-[12px] text-[16px] rounded-md',
  lg: 'px-xl py-md text-[16px] rounded-md',
}

const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  ({ variant = 'primary', size = 'md', className, ...props }, ref) => {
    return (
      <button
        ref={ref}
        className={cn(
          'inline-flex items-center justify-center font-body font-bold',
          'transition-all duration-150 ease-out',
          'focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-interactive-focus',
          'disabled:pointer-events-none disabled:opacity-50',
          variantClasses[variant],
          sizeClasses[size],
          className,
        )}
        {...props}
      />
    )
  }
)

Button.displayName = 'Button'

export { Button }
export type { ButtonProps, ButtonVariant, ButtonSize }
```

Key implementation details per UI-SPEC:
- Primary variant: `#F5F5F5` bg, `#0D0D0D` text, hover reduces bg opacity to 85%
- Ghost variant: transparent bg, `#F5F5F5` text, `1px solid rgba(255,255,255,0.20)` border, hover border at 60%
- Destructive variant: `#EF4444` bg, white text, hover `#DC2626`
- Size sm: `px-16 py-8` (using spacing tokens px-md py-sm), 14px text, border-radius 4px (rounded-sm)
- Size md: `px-24 py-12` (using px-lg and arbitrary py-[12px] since 12px is not a token), 16px text, border-radius 8px (rounded-md)
- Size lg: `px-32 py-16` (using spacing tokens px-xl py-md), 16px text, border-radius 8px (rounded-md)
- Focus ring: `outline: 2px solid rgba(255,255,255,0.80)`, `outline-offset: 2px` (per UI-SPEC)
- Transition: `transition-all duration-150 ease-out` on background, border-color, opacity
- Uses `forwardRef` for composability with future Motion/GSAP wrappers
- Exports named `Button` plus types `ButtonProps`, `ButtonVariant`, `ButtonSize`
