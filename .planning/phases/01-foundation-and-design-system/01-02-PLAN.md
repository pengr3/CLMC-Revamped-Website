**Part B: Create LenisProvider**

Create `components/providers/LenisProvider.tsx` as a `"use client"` component:

```tsx
'use client'

import { useEffect, useRef } from 'react'
import Lenis from 'lenis'
import { usePathname } from 'next/navigation'

export function LenisProvider({ children }: { children: React.ReactNode }) {
  const lenisRef = useRef<Lenis | null>(null)
  const pathname = usePathname()

  useEffect(() => {
    // Respect prefers-reduced-motion (DSGN-04)
    const motionQuery = window.matchMedia('(prefers-reduced-motion: reduce)')
    if (motionQuery.matches) return // Skip Lenis entirely when motion is reduced

    const lenis = new Lenis({
      autoRaf: true,
      duration: 1.2,
      easing: (t: number) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      touchMultiplier: 2,
      syncTouch: false,
    })
    lenisRef.current = lenis

    return () => {
      lenis.destroy()
      lenisRef.current = null
    }
  }, [])

  // Scroll to top on route change
  useEffect(() => {
    lenisRef.current?.scrollTo(0, { immediate: true })
  }, [pathname])

  return <>{children}</>
}
```

Key details per RESEARCH.md Pattern 3:
- `autoRaf: true` — no manual requestAnimationFrame loop needed
- `duration: 1.2` — smooth but responsive scroll feel
- Exponential easing for natural deceleration
- `syncTouch: false` — do not smooth-scroll on touch devices (preserves native feel)
- Scrolls to top on route change
- DSGN-04 compliance: checks `prefers-reduced-motion` and skips Lenis initialization entirely when motion is reduced
- Clean up on unmount: `lenis.destroy()`
